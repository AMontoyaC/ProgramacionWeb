create table tipoProducto
(
	id_tipo int auto_increment,
	nombre varchar(50),
	CONSTRAINT pk_tipoProducto primary key (id_tipo)
)
engine=myisam;

create table Productos
(
	id_producto int auto_increment,
	nombre varchar(25),
	descripcion varchar(200),
	precio decimal,
	cantidad int,
	CONSTRAINT pk_productos primary key (id_producto)
)
engine=myisam;

create table Clientes
(
	id_cliente int auto_increment,
	nombre varchar(20),
	apepat varchar(20),
	apemat varchar(20),
	email varchar(30),
	direccion varchar(50),
	ciudad varchar(20),
	cp char(5),
	telefono int,
	CONSTRAINT pk_Clientes primary key (id_cliente)
)
engine=myisam;

create table Usuario
(
	id_usuario int auto_increment,
	usuario varchar(20),
	password varchar(32),
	id_cliente int,
	CONSTRAINT pk_Usuario primary key (id_usuario),
	CONSTRAINT fk_Usuario foreign key (id_cliente) references Clientes (id_cliente)
)
engine=myisam;

create table Canasta
(
	id_canasta int auto_increment,
	cantidad int,
	total decimal,
	id_producto int,
	id_cliente int,
	CONSTRAINT pk_Canasta primary key (id_canasta),
	CONSTRAINT fk_Canasta1 foreign key (id_producto) references Productos (id_producto),
	CONSTRAINT fk_Canasta2 foreign key (id_cliente) references Clientes (id_cliente)
)
engine=myisam;


create table Pedido
(
	id_pedido int auto_increment,
	fecha date,
	id_canasta int,
	CONSTRAINT pk_Pedido primary key (id_pedido),
	CONSTRAINT fk_Pedido foreign key (id_canasta) references Canasta (id_canasta)
)
engine=myisam;
